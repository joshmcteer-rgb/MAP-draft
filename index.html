First-Time Setup Action Required

Your Supabase project is active, but we need to create the table structure and grant public permissions. Run this code inside your **Supabase SQL Editor** to construct the schema: 

-- 1. Create evaluation roadmap table
CREATE TABLE IF NOT EXISTS evaluations (
  id TEXT PRIMARY KEY,
  prospect_name TEXT NOT NULL,
  discovery_date DATE,
  go_live_date DATE,
  custom_milestones JSONB DEFAULT '[]'::JSONB,
  anchor_overrides JSONB DEFAULT '{}'::JSONB,
  anchor_statuses JSONB DEFAULT '{}'::JSONB,
  roi JSONB DEFAULT '{}'::JSONB,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. DISABLE RLS entirely to allow public writes/reads (highly recommended for evaluation sandbox)
ALTER TABLE evaluations DISABLE ROW LEVEL SECURITY;

-- 3. Enable Postgres realtime settings
ALTER TABLE evaluations REPLICA IDENTITY FULL;
ALTER PUBLICATION supabase_realtime ADD TABLE evaluations;
